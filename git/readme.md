# git 配置
  github, gitee 身份
- git config --global user.name "用户名"
- git config --global user.email "邮箱"
- git config --list 查看配置信息
- git config --global core.editor "vim" 配置编辑器为vim
- git config --global core.editor "code --wait" 配置编辑器为vscode
